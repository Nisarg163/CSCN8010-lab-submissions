from tensorflow.keras.models import Sequential
from tensorflow.keras.layers import Flatten, Dense

# Normalize pixel values to be between 0 and 1
x_train, x_test = x_train / 255.0, x_test / 255.0

# Build the neural network
model = Sequential([
    Flatten(input_shape=(28, 28)),  # Flatten the 28x28 images to a 1D array
    Dense(128, activation='relu'),
    Dense(10, activation='softmax')  # 10 output classes
])

# Compile the model
model.compile(optimizer='adam', loss='sparse_categorical_crossentropy', metrics=['accuracy'])

# Train the model
model.fit(x_train, y_train, epochs=10, validation_data=(x_test, y_test))

# Evaluate the model on the test set
test_loss, test_acc = model.evaluate(x_test, y_test)
print(f"\nTest Accuracy: {test_acc}")

# Predictions
predictions = model.predict(x_test)
predicted_classes = np.argmax(predictions, axis=1)

# Confusion Matrix
confusion_matrix = tf.math.confusion_matrix(y_test, predicted_classes)

# Visualize Confusion Matrix
import seaborn as sns
plt.figure(figsize=(8, 6))
sns.heatmap(confusion_matrix, annot=True, fmt='d', cmap='Blues', xticklabels=np.unique(y_test), yticklabels=np.unique(y_test))
plt.xlabel('Predicted')
plt.ylabel('True')
plt.show()

# Explore incorrect classifications
incorrect_indices = np.where(predicted_classes != y_test)[0]
print(f"Number of incorrect classifications: {len(incorrect_indices)}")

# Visualize some examples of incorrect classifications
plt.figure(figsize=(12, 12))
for i in range(9):
    plt.subplot(3, 3, i + 1)
    plt.imshow(x_test[incorrect_indices[i]], cmap='gray')
    plt.title(f"True: {y_test[incorrect_indices[i]]}, Predicted: {predicted_classes[incorrect_indices[i]]}")
    plt.axis('off')
plt.show()
